1. Neutral Reviewers
select reviewers.reviewer_name from ratings inner join reviewers ON ratings.reviewer_id=reviewers.reviewer_id 
WHERE ratings.reviewer_stars is NULL; 


2. Movie Character

SELECT CONCAT(directors.director_first_name, directors.director_last_name) AS director_name,movie_title 
FROM movies
INNER JOIN movies_directors as mvd ON mvd.movie_id=movies.movie_id
INNER JOIN directors ON directors.director_id=mvd.director_id
INNER JOIN movies_cast ON movies_cast.movie_id=movies.movie_id AND movies_cast.role='SeanMaguire';


3. short films

SELECT t1.movie_title, t1.movie_year, CONCAT(directors.director_first_name, directors.director_last_name) 
AS director_name, CONCAT(actors.actor_first_name, actors.actor_last_name) AS actor_name, 
movies_cast.role AS role

FROM (SELECT movie_id, movie_title, movie_year, movie_time FROM movies ORDER BY movies.movie_time ASC) AS t1

INNER JOIN movies_directors as mvd ON mvd.movie_id=t1.movie_id

INNER JOIN directors ON directors.director_id=mvd.director_id

INNER JOIN movies_cast ON movies_cast.movie_id=t1.movie_id

INNER JOIN actors ON actors.actor_id=movies_cast.actor_id

ORDER BY t1.movie_time ASC;


4. Actors and their movies

SELECT movie_title FROM movies as m 
INNER JOIN movies_cast as mc ON m.movie_id=mc.movie_id 
WHERE mc.actor_id IN (SELECT actor_id FROM movies_cast GROUP BY actor_id HAVING count(movie_id)>=2);
